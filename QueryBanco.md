```sql
CREATE DATABASE ParcialDB

USE parcialdb

CREATE TABLE MARCA( 
 ID int NOT NULL,
 Nome VARCHAR(50),
 CONSTRAINT PK_MARCA PRIMARY KEY (ID) 
)

CREATE TABLE CARRO(
 ID int NOT NULL, 
 ID_MARCA int NOT NULL,
 Nome varchar(50),
 Velocidade int,
 Obs varchar(50),
 CONSTRAINT FK_MARCA_CARRO FOREIGN KEY (ID) REFERENCES MARCA(id),
 CONSTRAINT PK_CARRO PRIMARY KEY (ID) 
)

CREATE TABLE CLIENTE(
 ID int NOT NULL, 
 Nome varchar(50),
 Titulo_Eleitor int,
 CPF varchar(50),
 CONSTRAINT PK_CLIENTE PRIMARY KEY (ID) 
)

CREATE TABLE VENDEDOR(
 ID int NOT NULL, 
 Nome varchar(50),
 CONSTRAINT PK_VENDEDOR PRIMARY KEY (ID) 
)

CREATE TABLE VENDAS(
 ID int NOT NULL,
 ID_CLIENTE int NOT NULL, 
 ID_VENDEDOR int NOT NULL,
 ID_CARRO int NOT NULL,
 Valor varchar(25),
 CONSTRAINT FK_CLIENTE_VENDAS FOREIGN KEY (ID) REFERENCES CLIENTE(id),
 CONSTRAINT FK_CARRO_VENDAS FOREIGN KEY (ID) REFERENCES CARRO(id),
 CONSTRAINT FK_VENDEDOR_VENDAS FOREIGN KEY (ID) REFERENCES VENDEDOR(id),
 CONSTRAINT PK_VENDAS PRIMARY KEY (ID) 
)
```
